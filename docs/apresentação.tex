\begin{frame}{Introdução e Escopo}

        \item \textbf{Fontes de Dados:}

\begin{frame}{Modelagem: Grafo Bipartido}
    O sistema é modelado como um grafo $G = (P, I, E)$:
    \begin{description}
        \item[Vértices Pratos ($P$):] Representam os itens do cardápio.
        \item[Vértices Ingredientes ($I$):] Representam os componentes atômicos.
        \item[Arestas ($E$):] Conexão binária indicando que um ingrediente compõe o prato.
    \end{description}
    \vspace{0.5cm}
    \textbf{Por que Bipartido?} Não existem conexões diretas $P \leftrightarrow P$ ou $I \leftrightarrow I$, permitindo focar na vizinhança comum para o cálculo de similaridade.
\end{frame}

\begin{frame}{Métrica: Índice de Jaccard}
    A similaridade técnica entre dois pratos $u$ e $v$ é quantificada matematicamente:
    \[ J(u,v) = \frac{|N(u) \cap N(v)|}{|N(u) \cup N(v)|} \]
    \begin{itemize}
        \item $N(u) \cap N(v)$: Ingredientes compartilhados (Interseção).
        \item $N(u) \cup N(v)$: Repertório total de ingredientes (União).
    \end{itemize}
    \textbf{Aplicação:} Identificar pratos "espelhados" que, apesar de nomes diferentes, compartilham a mesma base técnica.
\end{frame}
